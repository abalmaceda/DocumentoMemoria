%   1: ruta imagen
%   2: medida
%   3: caption
%   4: label
%Ej: \figura{portada.png}{width=10cm}{este es el texto q va abajo de la figura}{fig:portada}
\newcommand{\figura}[4]
{
   \begin{figure}[H]
     \centering
     \caption{#3}\label{#4}
     \includegraphics[#2]{#1}
   \end{figure}
}
